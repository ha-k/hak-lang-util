import hlt.language.util.ArrayList;

public class ArrayListTest
{

    {
      ArrayList s = new ArrayList();

      System.out.println(s);

      s.add("1");
      s.add("2");
      s.add("3");
      s.add("1");
      s.add("5");

      System.out.println(s);

      for (Iterator i = s.iterator(); i.hasNext();)
        System.out.println(i.next());

      for (int i=s.size(); i-->0;)
